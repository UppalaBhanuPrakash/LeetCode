class Solution {
public:
    int minDeletionSize(vector<string>& strs) {
        set<int> st;
        for(int i=0;i<strs.size()-1;i++){
            for(int j=0;j<strs[i].size();j++){
                if(strs[i][j]>strs[i+1][j]){
                st.insert(j);}
            }
        }
        return st.size();
    }
};
