class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        vector<int> v;
        map<int,int> mp;
        int tar;
        for(int i=0;i<nums.size();i++){
            tar=target-nums[i];
            if(mp.find(tar)!=mp.end()){
                v.push_back(mp[tar]);
                v.push_back(i);
                break;
            }
            mp[nums[i]]=i;
        }
        return v;
    }
};
